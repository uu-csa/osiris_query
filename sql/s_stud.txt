select
    s_sih.STUDENTNUMMER,
    s_stud.NATIONALITEIT,
    s_stud.NATIONALITEIT2,
    s_stud.BURGERSERVICENUMMER,
    s_stud.SL_INGELOGD_DIGID
from
    OSIRIS.OST_STUDENT_INSCHRIJFHIST s_sih
    join OSIRIS.OST_STUDENT s_stud
    on s_sih.STUDENTNUMMER = s_stud.STUDENTNUMMER
where
    s_sih.COLLEGEJAAR = [collegejaar]
    and s_sih.EXAMENTYPE_CSA <> 'Q'
    and s_sih.SOORT_INSCHRIJVING_FAC = 'S'
