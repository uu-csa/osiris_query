select
    s_ih.STUDENTNUMMER,
    s_stu.NATIONALITEIT,
    s_stu.NATIONALITEIT2,
    s_stu.BURGERSERVICENUMMER,
    s_stu.SL_INGELOGD_DIGID
from
    OSIRIS.OST_STUDENT_INSCHRIJFHIST s_ih
    join OSIRIS.OST_STUDENT s_stu
    on s_ih.STUDENTNUMMER = s_stu.STUDENTNUMMER
where
    s_ih.COLLEGEJAAR = [collegejaar]
    and s_ih.EXAMENTYPE_CSA <> 'Q'
    and s_ih.SOORT_INSCHRIJVING_FAC = 'S'
