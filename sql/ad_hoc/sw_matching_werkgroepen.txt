select
    s_cur_wrk.STUDENTNUMMER,
    s_cur_wrk.GROEPSNUMMER,
    s_cur_wrk.WERKVORM,
    s_cur_wrk.AANVANGSBLOK,
    s_cur_wrk.COLLEGEJAAR,
    s_cur_wrk.CURSUS
from
    OSIRIS.OST_STUDENT_CURSUS_WERKVORM s_cur_wrk
where
    s_cur_wrk.WERKVORM = "WERKGROEP"
    and s_cur_wrk.AANVANGSBLOK = "3"
    and s_cur_wrk.COLLEGEJAAR = 2018
    and (s_cur_wrk.CURSUS = "201200040"
    or s_cur_wrk.CURSUS = "201200038")
