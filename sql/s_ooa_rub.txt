select
    s_ooa_rub.IO_AANVR_ID,
    s_ooa_rub.IO_PROCES,
    s_ooa_rub.HOOFDSTUK,
    s_ooa_rub.STATUS
from
    OSIRIS.OST_STUDENT_IO_HOOFDSTUK s_ooa_rub
    join OSIRIS.OST_STUDENT_IO_IN_AANVRAAG s_ooa_aan
    on s_ooa_rub.IO_AANVR_ID = s_ooa_aan.IO_AANVR_ID
where
    s_ooa_aan.COLLEGEJAAR = [collegejaar]
    and s_ooa_aan.CROHO is not null
    and s_ooa_aan.CROHO < 70000
    and s_ooa_rub.STATUS is not null
