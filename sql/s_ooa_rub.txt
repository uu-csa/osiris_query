select
    s_rub.IO_AANVR_ID,
    s_rub.IO_PROCES,
    s_rub.HOOFDSTUK,
    s_rub.STATUS
from
    OSIRIS.OST_STUDENT_IO_HOOFDSTUK s_rub
    join OSIRIS.OST_STUDENT_IO_IN_AANVRAAG s_ooa
    on s_rub.IO_AANVR_ID = s_ooa.IO_AANVR_ID
where
    s_ooa.COLLEGEJAAR = [collegejaar]
    and s_ooa.CROHO is not null
    and s_ooa.CROHO < 70000
    and s_rub.STATUS is not null
