select
    OST_STUDENT_INSCHRIJFHIST.STUDENTNUMMER,
    OST_STUDENT_INSCHRIJFHIST.OPLEIDING,
    OST_STUDENT.VERBLIJFSDOCUMENT
from
    OSIRIS.OST_STUDENT inner join OSIRIS.OST_STUDENT_INSCHRIJFHIST
    on OST_STUDENT.STUDENTNUMMER = OST_STUDENT_INSCHRIJFHIST.STUDENTNUMMER
where
    OST_STUDENT_INSCHRIJFHIST.ACTIEFCODE_OPLEIDING_CSA=4
    and OST_STUDENT_INSCHRIJFHIST.COLLEGEJAAR = 2018
    and OST_STUDENT_INSCHRIJFHIST.INGANGSDATUM = to_date('01-09-2018', 'DD-MM-YYYY')
    and OST_STUDENT.NATIONALITEIT <> 'NL'
    and OST_STUDENT.VERBLIJFSDOCUMENT is not null
    and OST_STUDENT.VERBLIJFSDOCUMENT <> 'IN'
