select
    OST_STUDENT_INSCHRIJFHIST.STUDENTNUMMER,
    OST_STUDENT_INSCHRIJFHIST.OPLEIDING,
    OST_STUDENT_INSCHRIJFHIST.CROHO,
    OST_STUDENT_INSCHRIJFHIST.COLLEGEJAAR,
    OST_STUDENT_INSCHRIJFHIST.EXAMENTYPE_CSA,
    OST_STUDENT_INSCHRIJFHIST.INGANGSDATUM,
    OST_STUDENT_OPLEIDING.AANVANGSJAAR_OPLEIDING
from
    OSIRIS.OST_STUDENT_INSCHRIJFHIST join OSIRIS.OST_STUDENT_OPLEIDING
    on OST_STUDENT_INSCHRIJFHIST.OPLEIDING = OST_STUDENT_OPLEIDING.OPLEIDING
    and OST_STUDENT_INSCHRIJFHIST.STUDENTNUMMER = OST_STUDENT_OPLEIDING.STUDENTNUMMER
where
    OST_STUDENT_INSCHRIJFHIST.ACTIEFCODE_OPLEIDING_CSA=0
    and OST_STUDENT_INSCHRIJFHIST.COLLEGEJAAR <= 2018
    and OST_STUDENT_INSCHRIJFHIST.INGANGSDATUM <= to_date('01-09-2018', 'DD-MM-YYYY')
    and OST_STUDENT_INSCHRIJFHIST.SOORT_INSCHRIJVING_FAC = 'S'
