select
    sih.STUDENTNUMMER,
    sih.OPLEIDING,
    sih.CROHO,
    opl.FACULTEIT,
    sih.COLLEGEJAAR,
    sih.EXAMENTYPE_CSA,
    sih.INGANGSDATUM,
    sop.AANVANGSJAAR_OPLEIDING
from
    OSIRIS.OST_STUDENT_INSCHRIJFHIST sih
    join OSIRIS.OST_STUDENT_OPLEIDING sop
    on sih.OPLEIDING = sop.OPLEIDING
    and sih.STUDENTNUMMER = sop.STUDENTNUMMER
    join OSIRIS.OST_OPLEIDING opl
    on sih.OPLEIDING = opl.OPLEIDING
where
    sih.ACTIEFCODE_OPLEIDING_CSA = 0
    and sih.COLLEGEJAAR <= 2018
    and sih.INGANGSDATUM <= to_date('01-09-2018', 'DD-MM-YYYY')
    and sih.SOORT_INSCHRIJVING_FAC = 'S'
