select distinct
    fr.STUDENTNUMMER,
    jp.COLLEGEGELDTYPE,
    jp.SOORT_COLLEGEGELD,
    ct.OMSCHRIJVING,
    jp.NUMMER_FIN_REGEL
from
    OSIRIS.OST_JOURNAALPOST jp
    inner join OSIRIS.OST_STUDENT_FIN_REGEL fr
    on jp.NUMMER_FIN_REGEL = fr.NUMMER_FIN_REGEL
    inner join OSIRIS.OST_DEB_COLLEGEGELDTYPE ct
    on jp.COLLEGEGELDTYPE = ct.COLLEGEGELDTYPE
where
    jp.COLLEGEJAAR = 2018
    and jp.MAAND = 9
